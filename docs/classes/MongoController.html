import { ApiBase } from "../ApiBase"
import * as mongoose from 'mongoose';
import { ClientSession } from "mongoose";

export class MongoController {

    public async connect(urlConection: string): Promise<void> {
        try {
            mongoose.connect(urlConection)
            const db = mongoose.connection;

            db.on("error", event => {
                ApiBase.logException(`connection error: ${event}`);
            });
            db.once("open", function () {
                ApiBase.log("Connected successfully", 'MongoDb');
            });
            db.on('serverClosed', event => {
                ApiBase.log('received serverClosed', 'MongoDb');
                ApiBase.log(JSON.stringify(event, null, 2));
            });
            db.on('topologyClosed', event => {
                ApiBase.log('received topologyClosed', 'MongoDb');
                ApiBase.log(JSON.stringify(event, null, 2));
            });
            db.on('serverHeartbeatFailed', event => {
                ApiBase.log('received serverHeartbeatFailed', 'MongoDb');
                ApiBase.log(JSON.stringify(event, null, 2));
            });

        } catch (ex) {
            throw ApiBase.exception(ex);
        }
    }

    public isConnectionUp(): boolean {
        try {
            return (mongoose.connection && mongoose.connection['_readyState'] === 1);
        } catch (ex) {
            throw ApiBase.exception(ex);
        }
    }

    public async startTransaction(): Promise<ClientSession> {
        const session = await mongoose.startSession();
        if (!session) throw new Error(`Failed to start mongodb session`);

        session.startTransaction();
        return session;
    }

    public async abortTransaction(session: ClientSession): Promise<void> {
        if (!session) throw new Error('mongodb session is NULL');

        await session.abortTransaction();
        session.endSession();
    }

    public async commitTransaction(session: ClientSession): Promise<void> {
        if (!session) throw new Error('mongodb session is NULL');

        await session.commitTransaction();
        session.endSession();
    }

}
