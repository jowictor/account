import { ApiBase } from '../ApiBase';
import { Model, QueryOptions, InsertManyOptions } from 'mongoose';
import * as mongoose from 'mongoose';

export class ApiRepository {
	private readonly dbInstance: Model<unknown, {}, {}, {}>;
	private readonly modelName: string;
	constructor(modelName: string, schema: mongoose.Schema) {
		this.dbInstance = mongoose.model(modelName, schema);
		this.modelName = modelName;
	}

	protected updateQueryFactory(data: any): object {
		let updateQuery = {};
		updateQuery = Object.assign(updateQuery, data);
		if (updateQuery.hasOwnProperty('_id')) delete updateQuery['_id'];
		return updateQuery;
	}

	protected async find(filter: object, _projection?: object, _options?: QueryOptions | null, _timeout?: number): Promise<ApiBase.ApiResponse> {
		try {
			if (!filter) return ApiBase.error('Missing filter', ApiBase.Status.ERROR_NULL_PARAMETER);

			const projection = _projection ? _projection : {};
			const options = _options ? _options : {};

			const result = await this.dbInstance.find(filter, projection, options);
			if (result && result.length > 0) return ApiBase.success(result);

			return ApiBase.error('No content found', ApiBase.Status.ERROR_NOT_FOUND);
		} catch (ex) {
			throw ApiBase.exception(ex);
		}
	}

	protected async insert(data: object | any[], _options?: InsertManyOptions): Promise<ApiBase.ApiResponse> {
		try {
			if (!data) return ApiBase.error('Missing data params', ApiBase.Status.ERROR_NULL_PARAMETER);

			const content = this.updateQueryFactory(data);
			const options = (_options) ? _options : {};
			const result = await this.dbInstance.insertMany(content, options);
			if (!result) {
				return ApiBase.error(`Failed to insert document on collection ${this.modelName}`, ApiBase.Status.FAILED);
			}
			return ApiBase.success(result[0]);
		} catch (ex) {
			if (ex.code && ex.code === 11000) return ApiBase.error(`${ex.message}`, ApiBase.Status.ERROR_CONFLICT);
			throw ApiBase.exception(new Error(ex));
		}
	}

	protected async update(filter: object, _updateQuery: object, options?: QueryOptions): Promise<ApiBase.ApiResponse> {
		try {
			if (!filter) return ApiBase.error('Missing filter params to update', ApiBase.Status.ERROR_NULL_PARAMETER);
			if (!_updateQuery) return ApiBase.error('Missing updateQuery params to update', ApiBase.Status.ERROR_NULL_PARAMETER);

			const updateQuery = this.updateQueryFactory(_updateQuery);

			const result = await this.dbInstance.updateOne(filter, updateQuery, options);
			if (result && result.matchedCount > 0 && result.modifiedCount > 0) return ApiBase.success(true);

			return ApiBase.error(`Failed to update document on collection ${this.modelName}`, ApiBase.Status.FAILED);
		} catch (ex) {
			return ApiBase.exception(ex);
		}
	}

	protected async updateById(id: string | object, _updateQuery: object, options?: object): Promise<ApiBase.ApiResponse> {
		try {
			if (!id) return ApiBase.error('Missing id params to update', ApiBase.Status.ERROR_NULL_PARAMETER);
			if (!_updateQuery) return ApiBase.error('Missing updateQuery params to update', ApiBase.Status.ERROR_NULL_PARAMETER);

			const updateQuery = this.updateQueryFactory(_updateQuery);

			const result = await this.dbInstance.findByIdAndUpdate(id, updateQuery, options);
			if (result) return ApiBase.success(true);

			return ApiBase.error(`Failed to update document on collection ${this.modelName} - ID: ${id}`, ApiBase.Status.FAILED);
		} catch (ex) {
			return ApiBase.exception(ex);
		}
	}

	protected async findOneAndUpdateByQuery(filter: any, query: any, _timeout?: number, collectionName?: string, _options?: any): Promise<ApiBase.ApiResponse> {
		try {
			if (!filter) return ApiBase.error('Missing id params to update', ApiBase.Status.ERROR_NULL_PARAMETER);
			if (!query) return ApiBase.error('Missing updateQuery params to update', ApiBase.Status.ERROR_NULL_PARAMETER);
			const result = await this.dbInstance.findOneAndUpdate(filter, query, { new: true });
			if (result) return ApiBase.success(true);

			return ApiBase.error(`Failed to update document on collection ${this.modelName} - Query: ${query}`, ApiBase.Status.FAILED);
		} catch (ex) {
			return ApiBase.exception(ex);
		}
	}

	protected async delete(id: string | object): Promise<ApiBase.ApiResponse> {
		try {
			if (!id) return ApiBase.error('Missing id params to delete', ApiBase.Status.ERROR_NULL_PARAMETER);

			const result = await this.dbInstance.deleteOne({ _id: id });
			if (!result) {
				return ApiBase.error(`Failed to delete document on collection ${this.modelName}`, ApiBase.Status.FAILED);
			}
			if (result) return ApiBase.success(true);
		} catch (ex) {
			return ApiBase.exception(ex);
		}
	}
}
